```c
// scheduler.c
static task_t scheduler_task[] = {
    {uart_rx_task, 5, 0},          // 5ms - UART数据接收
    {rfid_task, 5, 0},             // 5ms - RFID帧轮询
    {matrix_key_scan, 10, 0},      // 10ms - 矩阵键盘扫描
    {door_control_update, 100, 0}, // 100ms - 门禁超时检查
    {auth_manager_update, 100, 0}, // 100ms - 认证锁定超时检查 (新增)
    {user_admin_update, 500, 0},   // 500ms - 管理模式超时检查 (新增)
};

void scheduler_run(void) {
    for (uint8_t i = 0; i < task_num; i++) {
        uint32_t now_time = TIM_Get_MillisCounter();
        // 检查是否到达执行周期
        if (now_time >= scheduler_task[i].rate_ms + scheduler_task[i].last_run) {
            scheduler_task[i].last_run = now_time;
            scheduler_task[i].task_func();  // 执行任务
        }
    }
}
```

## 2. 用户数据库模块实现

### 2.1 Flash存储结构

```c
// user_database.h 关键定义
#define USER_DB_FLASH_BASE      0x08046000  // Flash起始地址
#define USER_DB_MAX_USERS       50          // 最大用户数
#define USER_ENTRY_SIZE         32          // 每用户32字节
#define USER_DB_MAGIC           0x55AA1234  // 魔数标识

// Flash布局:
// [0x08046000] Header (32 bytes)
// [0x08046020] User[0] (32 bytes)
// [0x08046040] User[1] (32 bytes)
// ...
// [0x08046020 + 49*32] User[49]
```

### 2.2 初始化逻辑

```c
// user_database.c
user_db_result_t user_db_init(void) {
    // 步骤1: 从Flash读取头部到内存缓存
    flash_read(USER_DB_HEADER_ADDR, (uint8_t *)&s_db_header, sizeof(user_db_header_t));

    // 步骤2: 检查魔数判断数据有效性
    if (s_db_header.magic != USER_DB_MAGIC) {
        // 魔数无效，需要格式化
        printf("[UserDB] Invalid magic, formatting database...\r\n");
        return user_db_format();  // 执行格式化
    }

    // 步骤3: 魔数有效，标记初始化完成
    s_initialized = 1;
    printf("[UserDB] Database initialized, %d users\r\n", s_db_header.user_count);
    return USER_DB_OK;
}

user_db_result_t user_db_format(void) {
    // 步骤1: 初始化头部结构
    memset(&s_db_header, 0, sizeof(user_db_header_t));
    s_db_header.magic = USER_DB_MAGIC;
    s_db_header.version = 0x0100;
    s_db_header.user_count = 0;
    s_db_header.next_id = 1;  // ID从1开始
    // 默认主密码: 0000
    s_db_header.master_password[0] = 0;
    s_db_header.master_password[1] = 0;
    s_db_header.master_password[2] = 0;
    s_db_header.master_password[3] = 0;

    // 步骤2: 写入头部到Flash
    write_header();

    // 步骤3: 擦除用户数据区
    flash_unlock();
    for (uint32_t addr = USER_DB_DATA_ADDR;
         addr < USER_DB_DATA_ADDR + USER_DB_MAX_USERS * USER_ENTRY_SIZE;
         addr += FLASH_PAGE_SIZE) {
        flash_erase_page(addr);
    }
    flash_lock();

    s_initialized = 1;
    return USER_DB_OK;
}
```

### 2.3 用户查找算法

```c
// 通过RFID查找用户 - 线性遍历
user_db_result_t user_db_find_by_rfid(const uint8_t uid[4], user_entry_t *out) {
    user_entry_t user;

    // 遍历所有槽位
    for (uint16_t i = 0; i < USER_DB_MAX_USERS; i++) {
        // 从Flash读取用户数据
        read_user(i, &user);

        // 检查用户是否有效
        if (!(user.flags & USER_FLAG_VALID)) {
            continue;  // 跳过无效槽位
        }

        // 检查是否已录入RFID
        if (!(user.flags & USER_FLAG_RFID_EN)) {
            continue;  // 跳过未录入RFID的用户
        }

        // 比较RFID UID (4字节)
        if (memcmp(user.rfid_uid, uid, 4) == 0) {
            *out = user;  // 找到匹配

            // 检查用户是否被禁用
            if (user.flags & USER_FLAG_DISABLED) {
                return USER_DB_ERR_DISABLED;
            }
            return USER_DB_OK;
        }
    }
    return USER_DB_ERR_NOT_FOUND;
}

// 从Flash读取用户数据
static void read_user(uint16_t slot, user_entry_t *user) {
    uint32_t addr = USER_DB_DATA_ADDR + slot * USER_ENTRY_SIZE;
    // 直接从Flash地址读取
    flash_read(addr, (uint8_t *)user, USER_ENTRY_SIZE);
}
```

### 2.4 Flash写入逻辑

```c
// 写入用户到Flash
static user_db_result_t write_user(uint16_t slot, const user_entry_t *user) {
    uint32_t addr = USER_DB_DATA_ADDR + slot * USER_ENTRY_SIZE;
    uint32_t *p = (uint32_t *)user;

    flash_unlock();

    // CH32V30x Flash特性: 写入前必须擦除
    // 擦除最小单位是256字节页
    flash_erase_page(addr);

    // 以4字节(word)为单位写入
    for (int i = 0; i < (USER_ENTRY_SIZE / 4); i++) {
        FLASH_Status status = flash_write_word(addr + i * 4, p[i]);
        if (status != FLASH_COMPLETE) {
            flash_lock();
            return USER_DB_ERR_FLASH;
        }
    }

    flash_lock();
    return USER_DB_OK;
}
```

### 2.5 密码验证逻辑

```c
// 验证用户密码
user_db_result_t user_db_verify_password(const uint8_t password[4], user_entry_t *out) {
    user_entry_t user;

    for (uint16_t i = 0; i < USER_DB_MAX_USERS; i++) {
        read_user(i, &user);

        // 跳过无效用户
        if (!(user.flags & USER_FLAG_VALID)) continue;
        // 跳过未设置密码的用户
        if (!(user.flags & USER_FLAG_PWD_EN)) continue;

        // 比较4位密码
        if (memcmp(user.password, password, 4) == 0) {
            *out = user;
            if (user.flags & USER_FLAG_DISABLED) {
                return USER_DB_ERR_DISABLED;
            }
            return USER_DB_OK;
        }
    }
    return USER_DB_ERR_NOT_FOUND;
}

// 验证主密码 (管理员密码)
user_db_result_t user_db_verify_master_password(const uint8_t password[4]) {
    // 直接与头部中的主密码比较
    if (memcmp(s_db_header.master_password, password, 4) == 0) {
        return USER_DB_OK;
    }
    return USER_DB_ERR_PASSWORD;
}
```

## 3. 开门日志模块实现

### 3.1 循环缓冲原理

```
日志存储采用循环缓冲(Ring Buffer)设计:

    ┌───┬───┬───┬───┬───┬───┬───┬───┐
    │ 0 │ 1 │ 2 │ 3 │ 4 │ 5 │ 6 │ 7 │  (简化为8条)
    └───┴───┴───┴───┴───┴───┴───┴───┘
          ↑                   ↑
        tail                head
        (最旧)              (最新)

写入新记录:
1. 写入 record[head]
2. head = (head + 1) % MAX_RECORDS
3. 如果 total < MAX: total++
   否则: tail = (tail + 1) % MAX_RECORDS  (覆盖最旧)

读取记录 (index=0 最新):
slot = (head - 1 - index + MAX) % MAX
```

### 3.2 日志写入实现

```c
// access_log.c
access_log_result_t access_log_record(uint16_t user_id,
                                       auth_method_t method,
                                       auth_result_t result,
                                       const uint8_t *uid) {
    access_record_t record;
    memset(&record, 0, sizeof(access_record_t));

    // 步骤1: 填充记录数据
    record.timestamp = TIM_Get_MillisCounter();  // 当前毫秒计数
    record.user_id = user_id;                     // 用户ID (0=未知)
    record.auth_method = (uint8_t)method;         // 认证方式
    record.result = (uint8_t)result;              // 认证结果
    if (uid != NULL) {
        memcpy(record.rfid_uid, uid, 4);          // RFID卡号
    }

    // 步骤2: 确定写入槽位 (当前head位置)
    uint16_t write_slot = s_log_header.head_index;

    // 步骤3: 写入Flash
    access_log_result_t ret = write_record(write_slot, &record);
    if (ret != ACCESS_LOG_OK) {
        return ret;
    }

    // 步骤4: 更新索引
    s_log_header.head_index = (s_log_header.head_index + 1) % ACCESS_LOG_MAX_RECORDS;
    s_log_header.sequence_number++;  // 全局序列号递增

    if (s_log_header.total_records < ACCESS_LOG_MAX_RECORDS) {
        // 未满：增加记录数
        s_log_header.total_records++;
    } else {
        // 已满：移动tail覆盖最旧记录
        s_log_header.tail_index = (s_log_header.tail_index + 1) % ACCESS_LOG_MAX_RECORDS;
    }

    // 步骤5: 更新头部到Flash (持久化索引)
    return write_header();
}
```

### 3.3 日志读取实现

```c
// 获取指定索引的记录 (0=最新, count-1=最旧)
access_log_result_t access_log_get_record(uint16_t index, access_record_t *out) {
    if (index >= s_log_header.total_records) {
        return ACCESS_LOG_ERR_INVALID;
    }

    // 计算实际槽位
    // head指向下一个写入位置，最新记录在head-1
    int16_t slot = (int16_t)s_log_header.head_index - 1 - (int16_t)index;

    // 处理负数情况 (循环)
    if (slot < 0) {
        slot += ACCESS_LOG_MAX_RECORDS;
    }

    read_record((uint16_t)slot, out);
    return ACCESS_LOG_OK;
}
```

## 4. 密码输入模块实现

### 4.1 状态机设计

```c
// password_input.c 内部状态
static struct {
    pwd_input_state_t state;           // 当前状态
    uint8_t password[4];               // 已输入的密码
    uint8_t input_count;               // 已输入位数
    uint32_t start_time;               // 开始输入时间
    pwd_input_callback_t callback;     // 完成回调
} s_pwd_input;

// 状态转换:
// IDLE --start()--> INPUTTING --4位完成--> COMPLETE
//                       |
//                       +--超时--> TIMEOUT
//                       |
//                       +--取消--> CANCELLED
```

### 4.2 按键处理逻辑

```c
void pwd_input_on_key(uint8_t key_id) {
    // 只在输入状态处理
    if (s_pwd_input.state != PWD_STATE_INPUTTING) {
        return;
    }

    uint8_t digit = 0xFF;  // 无效值

    // 键盘映射
    if (key_id >= 1 && key_id <= 9) {
        digit = key_id;        // 键1-9 = 数字1-9
    } else if (key_id == 10) {
        digit = 0;             // 键10 = 数字0
    } else if (key_id == 11) {
        // 退格
        if (s_pwd_input.input_count > 0) {
            s_pwd_input.input_count--;
            s_pwd_input.password[s_pwd_input.input_count] = 0;
        }
        return;
    } else if (key_id == 12) {
        // 取消
        pwd_input_cancel();
        return;
    }

    // 添加数字
    if (digit <= 9 && s_pwd_input.input_count < 4) {
        s_pwd_input.password[s_pwd_input.input_count] = digit;
        s_pwd_input.input_count++;

        // 检查是否输入完成
        if (s_pwd_input.input_count >= 4) {
            s_pwd_input.state = PWD_STATE_COMPLETE;

            // 触发回调
            if (s_pwd_input.callback) {
                s_pwd_input.callback(s_pwd_input.password,
                                      s_pwd_input.input_count,
                                      PWD_STATE_COMPLETE);
            }
        }
    }
}
```

## 5. 认证管理器实现

### 5.1 核心状态机

```c
// auth_manager.c 内部状态
static struct {
    auth_mgr_state_t state;        // IDLE/PROCESSING/PASSWORD_INPUT/LOCKED
    uint8_t fail_count;            // 失败次数
    uint32_t lockout_start;        // 锁定开始时间
    auth_result_callback_t callback;
    uint8_t last_rfid_uid[4];      // 用于日志
} s_auth_mgr;

// 状态转换:
// IDLE --刷卡/指纹/密码--> PROCESSING --完成--> IDLE
//   |                          |
//   +--启动密码输入--> PASSWORD_INPUT --密码完成--> PROCESSING
//                          |
// IDLE <--60秒超时-- LOCKED <--失败5次-- PROCESSING
```

### 5.2 RFID认证流程

```c
auth_result_t auth_process_rfid(const RFID_Frame *frame) {
    // 步骤1: 检查锁定状态
    if (s_auth_mgr.state == AUTH_MGR_LOCKED) {
        printf("[AuthMgr] System locked, rejecting RFID\r\n");
        return AUTH_RESULT_FAILED;
    }

    // 步骤2: 保存UID用于日志
    memcpy(s_auth_mgr.last_rfid_uid, frame->uid, 4);
    s_auth_mgr.state = AUTH_MGR_PROCESSING;

    // 步骤3: 在用户数据库中查找
    user_entry_t user;
    user_db_result_t db_ret = user_db_find_by_rfid(frame->uid, &user);

    // 步骤4: 根据结果处理
    if (db_ret == USER_DB_OK) {
        // 找到用户，认证成功
        auth_success(AUTH_RFID, &user, frame->uid);
        return AUTH_RESULT_SUCCESS;
    } else if (db_ret == USER_DB_ERR_DISABLED) {
        // 用户被禁用
        auth_failed(AUTH_RFID, AUTH_RESULT_USER_DISABLED, frame->uid);
        return AUTH_RESULT_USER_DISABLED;
    } else {
        // 未找到用户
        auth_failed(AUTH_RFID, AUTH_RESULT_USER_NOT_FOUND, frame->uid);
        return AUTH_RESULT_USER_NOT_FOUND;
    }
}
```

### 5.3 成功/失败处理

```c
// 认证成功处理
static void auth_success(auth_method_t method, const user_entry_t *user, const uint8_t *uid) {
    printf("[AuthMgr] Auth SUCCESS: user_id=%d, name=%s\r\n",
           user->id, user->name);

    // 1. 记录日志
    access_log_record(user->id, method, AUTH_RESULT_SUCCESS, uid);

    // 2. 重置失败计数
    s_auth_mgr.fail_count = 0;

    // 3. 开门 (5秒)
    door_control_unlock(method, AUTH_DOOR_OPEN_DURATION_MS);

    // 4. 播放成功语音
    BY8301_PlayIndex(1);

    // 5. 触发回调
    if (s_auth_mgr.callback) {
        s_auth_mgr.callback(method, AUTH_RESULT_SUCCESS, user);
    }

    s_auth_mgr.state = AUTH_MGR_IDLE;
}

// 认证失败处理
static void auth_failed(auth_method_t method, auth_result_t result, const uint8_t *uid) {
    printf("[AuthMgr] Auth FAILED: result=%d\r\n", result);

    // 1. 记录日志
    access_log_record(0, method, result, uid);

    // 2. 增加失败计数
    s_auth_mgr.fail_count++;

    // 3. 检查是否需要锁定
    if (s_auth_mgr.fail_count >= AUTH_MAX_FAIL_COUNT) {
        s_auth_mgr.state = AUTH_MGR_LOCKED;
        s_auth_mgr.lockout_start = TIM_Get_MillisCounter();
        printf("[AuthMgr] LOCKED! Too many failures\r\n");
        BY8301_PlayIndex(3);  // 警告语音
    } else {
        BY8301_PlayIndex(2);  // 失败语音
    }

    // 4. 触发回调
    if (s_auth_mgr.callback) {
        s_auth_mgr.callback(method, result, NULL);
    }

    if (s_auth_mgr.state != AUTH_MGR_LOCKED) {
        s_auth_mgr.state = AUTH_MGR_IDLE;
    }
}
```

### 5.4 锁定超时检查

```c

        printf("[Admin] ---- Admin Menu ----\r\n");
        printf("[Admin] 1=Add User  2=Delete User\r\n");
        printf("[Admin] 3=List Users  4=Change Master Pwd\r\n");
        printf("[Admin] 15=Exit\r\n");
        return 1;  // 成功
    }

    printf("[Admin] Invalid master password\r\n");
    return 0;  // 失败
}
```

### 6.3 添加用户流程

```c
void user_admin_start_add_user(void) {
    if (s_admin.state != ADMIN_STATE_UNLOCKED) return;

    s_admin.state = ADMIN_STATE_ADD_USER;
    s_admin.add_step = ADD_STEP_NAME;

    // 初始化新用户数据
    memset(&s_admin.new_user, 0, sizeof(user_entry_t));
    s_admin.password_set = 0;

    // 设置默认用户名
    snprintf(s_admin.new_user.name, 12, "User%d", user_db_get_count() + 1);

    s_admin.last_activity = TIM_Get_MillisCounter();
}

// RFID录入处理
void user_admin_on_rfid(const uint8_t uid[4]) {
    if (s_admin.state != ADMIN_STATE_ADD_USER) return;
    if (s_admin.add_step != ADD_STEP_RFID) return;

    // 检查RFID是否已被使用
    user_entry_t temp;
    if (user_db_find_by_rfid(uid, &temp) == USER_DB_OK) {
        printf("[Admin] RFID already registered to user %d\r\n", temp.id);
        return;  // 拒绝重复
    }

    // 保存RFID
    memcpy(s_admin.new_user.rfid_uid, uid, 4);
    s_admin.new_user.flags |= USER_FLAG_RFID_EN;

    // 进入下一步
    s_admin.add_step = ADD_STEP_FINGERPRINT;
    s_admin.last_activity = TIM_Get_MillisCounter();
}

// 确认添加
void user_admin_confirm(void) {
    if (s_admin.state == ADMIN_STATE_ADD_USER &&
        s_admin.add_step == ADD_STEP_CONFIRM) {
        finish_add_user();
    }
}

static void finish_add_user(void) {
    // 设置密码（如果有）
    if (s_admin.password_set) {
        memcpy(s_admin.new_user.password, s_admin.new_password, 4);
        s_admin.new_user.flags |= USER_FLAG_PWD_EN;
    }

    // 添加到数据库
    user_db_result_t ret = user_db_add_user(&s_admin.new_user);

    if (ret == USER_DB_OK) {
        printf("[Admin] User added: ID=%d\r\n", s_admin.new_user.id);
    }

    s_admin.add_step = ADD_STEP_DONE;
    s_admin.state = ADMIN_STATE_UNLOCKED;
}
```

```c
void user_admin_update(void) {
```

### 6.5 管理菜单按键处理 (v1.1.0)

```c
// user_admin.c - 按键处理核心逻辑
void user_admin_on_key(uint8_t key_id) {
    if (s_admin.state == ADMIN_STATE_LOCKED) return;
    update_activity();

    // 密码输入模式优先
    if (pwd_input_is_active()) {
        pwd_input_on_key(key_id);
        return;
    }

    switch (s_admin.state) {
        case ADMIN_STATE_UNLOCKED:
            // 管理菜单按键映射
            switch (key_id) {
                case 1:  user_admin_start_add_user();         break;
                case 2:  user_admin_start_delete_user();       break;
                case 3:  user_admin_start_list_users();        break;
                case 4:  user_admin_start_change_master_pwd(); break;
                case 12:
                case 15: user_admin_exit();                    break;
                default:
                    printf("[Admin] Menu: 1=Add 2=Del 3=List 4=ChgPwd 15=Exit\r\n");
                    break;
            }
            break;

        case ADMIN_STATE_ADD_USER:
            if (s_admin.add_step == ADD_STEP_PASSWORD) {
                pwd_input_start(add_user_password_callback);
                pwd_input_on_key(key_id);
            }
            break;

        case ADMIN_STATE_DELETE_USER:
            // 键1-9选择用户序号
            if (key_id >= 1 && key_id <= 9) {
                user_entry_t user;
                if (user_db_get_by_index(key_id - 1, &user) == USER_DB_OK) {
                    s_admin.selected_user_id = user.id;
                    printf("[Admin] Selected: ID=%d, Name=%s\r\n",
                           user.id, user.name);
                    printf("[Admin] Press 16(confirm) or 15(cancel)\r\n");
                }
            }
            break;

        case ADMIN_STATE_LIST_USERS:
            // 任意键返回菜单
            s_admin.state = ADMIN_STATE_UNLOCKED;
            break;

        default:
            break;
    }
}
```

**管理菜单数据流**:

```
键14 (正常模式)
  → auth_start_password()
  → 输入0000
  → password_complete_callback()
  → auth_process_password()
  → user_db_verify_master_password() ✓
  → user_admin_enter()
  → ADMIN_STATE_UNLOCKED + 打印菜单
  → 等待键1/2/3/4/15

键1 (管理菜单)
  → user_admin_start_add_user()
  → ADMIN_STATE_ADD_USER + ADD_STEP_NAME
  → 键16确认 → ADD_STEP_RFID
  → 刷卡 → ADD_STEP_FINGERPRINT
  → 键16跳过 → ADD_STEP_PASSWORD
  → 输入密码 → ADD_STEP_CONFIRM
  → 键16确认 → finish_add_user() → ADMIN_STATE_UNLOCKED
```

```c
// rfid_task.c 修改后的逻辑
void rfid_task(void) {
    RFID_Frame frame;
    RFID_Result ret = RFID_PollAutoFrame(&frame);

    if (ret == RFID_RET_OK) {
        uint32_t current_time = TIM_Get_MillisCounter();

        // 防抖：2秒内不重复处理
        if (current_time - s_last_card_time < RFID_DEBOUNCE_MS) {
            return;
        }
        s_last_card_time = current_time;

        printf("[RFID] Card: %02X %02X %02X %02X\r\n",
               frame.uid[0], frame.uid[1], frame.uid[2], frame.uid[3]);

        // 根据模式分发处理
        if (user_admin_is_active()) {
            // 管理模式：用于录入RFID
            user_admin_on_rfid(frame.uid);
        } else {
            // 正常模式：认证
            auth_process_rfid(&frame);
        }
    }
}
```
